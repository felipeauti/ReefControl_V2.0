export class SensorWidget extends HTMLElement { constructor() { super(); this.attachShadow({ mode: 'open' }); // Criar o template const template = document.createElement('template'); template.innerHTML = `
üå°Ô∏è
Sensor
--
Min: --
Max: --
Atualizado: --
`; this.shadowRoot.appendChild(template.content.cloneNode(true)); this._value = 0; this._unit = ''; this._status = 'offline'; this._min = Infinity; this._max = -Infinity; this._chartData = []; this._chart = null; } static get observedAttributes() { return ['value', 'unit', 'status', 'type']; } connectedCallback() { this._setupChart(); this._startUpdates(); } disconnectedCallback() { if (this._updateInterval) { clearInterval(this._updateInterval); } } attributeChangedCallback(name, oldValue, newValue) { if (oldValue === newValue) return; switch (name) { case 'value': this._updateValue(parseFloat(newValue)); break; case 'unit': this._updateUnit(newValue); break; case 'status': this._updateStatus(newValue); break; case 'type': this._updateType(newValue); break; } } _setupChart() { const ctx = this.shadowRoot.querySelector('.widget-chart canvas').getContext('2d'); this._chart = new Chart(ctx, { type: 'line', data: { labels: [], datasets: [{ data: [], borderColor: '#3498db', borderWidth: 2, fill: false, tension: 0.4 }] }, options: { responsive: true, maintainAspectRatio: false, scales: { x: { display: false }, y: { display: false } }, plugins: { legend: { display: false } }, elements: { point: { radius: 0 } } } }); } _updateValue(value) { this._value = value; this._min = Math.min(this._min, value); this._max = Math.max(this._max, value); this.shadowRoot.querySelector('.current-value').textContent = value.toFixed(1); this.shadowRoot.querySelector('.stat.min span').textContent = this._min.toFixed(1); this.shadowRoot.querySelector('.stat.max span').textContent = this._max.toFixed(1); this._updateChart(value); this._updateTimestamp(); } _updateUnit(unit) { this._unit = unit; this.shadowRoot.querySelector('.unit').textContent = unit; } _updateStatus(status) { this._status = status; const statusEl = this.shadowRoot.querySelector('.widget-status'); statusEl.className = 'widget-status ' + status; } _updateType(type) { const iconMap = { 'temp': 'üå°Ô∏è', 'ph': '‚öóÔ∏è', 'tds': 'üíß', 'level': 'üìè' }; const icon = this.shadowRoot.querySelector('.widget-icon'); icon.textContent = iconMap[type] || 'üìä'; } _updateChart(value) { const now = new Date(); this._chartData.push({ time: now, value: value }); // Manter apenas √∫ltimas 20 leituras if (this._chartData.length > 20) { this._chartData.shift(); } this._chart.data.labels = this._chartData.map(d => d.time.toLocaleTimeString()); this._chart.data.datasets[0].data = this._chartData.map(d => d.value); this._chart.update('none'); } _updateTimestamp() { const now = new Date(); this.shadowRoot.querySelector('.widget-update span').textContent = now.toLocaleTimeString(); } _startUpdates() { // Atualizar a cada 5 segundos this._updateInterval = setInterval(() => { this._fetchSensorData(); }, 5000); } async _fetchSensorData() { try { const response = await fetch('/api/sensors'); const data = await response.json(); const type = this.getAttribute('type'); if (data[type] !== undefined) { this.setAttribute('value', data[type]); this.setAttribute('status', 'online'); } } catch (error) { console.error('Erro ao atualizar sensor:', error); this.setAttribute('status', 'offline'); } } } // Registrar o componente customElements.define('sensor-widget', SensorWidget);
